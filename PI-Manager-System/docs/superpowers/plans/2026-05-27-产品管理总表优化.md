```python
def filter_products(self):
    """根据筛选条件过滤产品列表"""
    if not self._product_cache:
        return

    # 获取筛选条件
    customer_id = self.product_customer_filter.currentData()
    category_code = self.product_category_filter.currentData()
    search_keyword = self.product_search_input.text().strip().lower()

    # 过滤产品
    filtered = []
    for product in self._product_cache:
        # 客户筛选
        if customer_id and product.get('customer_id') != customer_id:
            continue

        # 类别筛选
        if category_code and product.get('category_id') != category_code:
            continue

        # 关键词搜索
        if search_keyword:
            match = False
            search_fields = [
                product.get('product_name', ''),
                product.get('detail_desc', ''),
                product.get('customer_model', ''),
                product.get('system_code', ''),
            ]
            # 搜索 OE 号
            for oe in product.get('oes', []) or []:
                search_fields.append(oe.get('oe_number', ''))
            # 搜索客户产品编号
            for code in product.get('codes', []) or []:
                search_fields.append(code.get('customer_code', ''))

            for field in search_fields:
                if field and search_keyword in str(field).lower():
                    match = True
                    break
            if not match:
                continue

        filtered.append(product)

    # 更新表格
    self._populate_products_table(filtered)

    # 更新状态
    total = len(self._product_cache)
    current = len(filtered)
    if total != current:
        self.product_status_label.setText(f"显示 {current}/{total} 条")
    else:
        self.product_status_label.setText(f"共 {total} 条")
```
